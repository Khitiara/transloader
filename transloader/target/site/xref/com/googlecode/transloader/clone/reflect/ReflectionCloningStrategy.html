9   /**
10   * A <code>CloningStrategy</code> that uses Java Reflection as its mechanism. Can clone whole object graphs or just
11   * necessary parts depending on how it is configured.
12   * 
13   * @author Jeremy Wales
14   */
15  public final class ReflectionCloningStrategy implements CloningStrategy {
16  	private final CyclicReferenceSafeTraverser cyclicReferenceSafeTraverser = new CyclicReferenceSafeTraverser();
17  
18  	private final CloningDecisionStrategy decider;
19  	private final InnerCloner arrayCloner;
20  	private final InnerCloner normalObjectCloner;
21  	private final CloningStrategy fallbackCloner;
22  
23  	/**
24  	 * Contructs a new <code>ReflectionCloningStrategy</code> with its dependencies injected.
25  	 * 
26  	 * @param cloningDecisionStrategy the strategy by which the decision to clone or not to clone a particular given
27  	 *            object is made
28  	 * @param instantiator the strategy by which to use instantiate normal objects (as opposed to arrays, for which
29  	 *            standard reflection is always adequate)
30  	 * @param fallbackCloningStrategy the <code>CloningStrategy</code> to fall back to when <code>this</code>
31  	 *            strategy fails
32  	 */
33  	public ReflectionCloningStrategy(CloningDecisionStrategy cloningDecisionStrategy,
34  			InstantiationStrategy instantiator, CloningStrategy fallbackCloningStrategy) {
35  		Assert.areNotNull(cloningDecisionStrategy, instantiator, fallbackCloningStrategy);
36  		decider = cloningDecisionStrategy;
37  		arrayCloner = new InnerArrayCloner(this);
38  		normalObjectCloner = new InnerNormalObjectCloner(this, instantiator);
39  		fallbackCloner = fallbackCloningStrategy;
40  	}
41  
42  	/**
43  	 * {@inheritDoc}
44  	 * <p>
45  	 * This implementation uses {@link CyclicReferenceSafeTraverser} to sucessfully handle cyclic references in the
46  	 * given object graph.
47  	 * </p>
48  	 * 
49  	 * @return a completely or partially cloned object graph, depending on the <code>CloningDecisionStrategy</code>
50  	 *         injected in
51  	 *         {@link #ReflectionCloningStrategy(CloningDecisionStrategy, InstantiationStrategy, CloningStrategy)},
52  	 *         with potentially the <code>original</code> itself being the top-level object in the graph returned if
53  	 *         it was not cloned
54  	 */
55  	public Object cloneObjectUsingClassLoader(final Object original, final ClassLoader targetClassLoader)
56  			throws Exception {
57  		Assert.areNotNull(original, targetClassLoader);
58  		Traversal cloningTraversal = new Traversal() {
59  			public Object traverse(Object currentObject, Map referenceHistory) throws Exception {
60  				return ReflectionCloningStrategy.this.clone(currentObject, targetClassLoader, referenceHistory);
61  			}
62  		};
63  		return cyclicReferenceSafeTraverser.performWithoutFollowingCircles(cloningTraversal, original);
64  	}
65  
66  	private Object clone(Object original, ClassLoader targetClassLoader, Map cloneHistory) throws Exception {
67  		if (original == null) return null;
68  		try {
69  			return performIntendedCloning(original, targetClassLoader, cloneHistory);
70  		} catch (Exception e) {
71  			return performFallbackCloning(original, targetClassLoader);
72  		}
73  	}
74  
75  	private Object performIntendedCloning(Object original, ClassLoader targetClassLoader, Map cloneHistory)
76  			throws Exception {
77  		InnerCloner innerCloner = original.getClass().isArray() ? arrayCloner : normalObjectCloner;
78  		Object clone = original;
79  		if (decider.shouldCloneObjectItself(original, targetClassLoader))
80  			clone = innerCloner.instantiateClone(original, targetClassLoader);
81  		cloneHistory.put(original, clone);
82  		if (decider.shouldCloneObjectContent(original, targetClassLoader))
83  			innerCloner.cloneContent(original, clone, targetClassLoader);
84  		return clone;
85  	}
86  
87  	private Object performFallbackCloning(Object original, ClassLoader targetClassLoader) throws Exception {
88  		return fallbackCloner.cloneObjectUsingClassLoader(original, targetClassLoader);
89  	}
