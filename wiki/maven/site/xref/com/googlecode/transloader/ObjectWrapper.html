10  /**
11   * The wrapper appropriate for wrapping around all <code>Object</code>s referencing <code>Class</code>es from
12   * potentially foreign <code>ClassLoader</code>s.
13   * 
14   * @author Jeremy Wales
15   */
16  public final class ObjectWrapper {
17  	private final Object wrappedObject;
18  	private final CloningStrategy cloner;
19  
20  	/**
21  	 * Constructs a new <code>ObjectWrapper</code> around the given object, which will use the given
22  	 * <code>CloningStrategy</code> when required. Note that using implementation of {@link Transloader} is the
23  	 * recommended way to produce these.
24  	 * 
25  	 * @param objectToWrap the object to wrap
26  	 * @param cloningStrategy the strategy for cloning
27  	 */
28  	public ObjectWrapper(Object objectToWrap, CloningStrategy cloningStrategy) {
29  		Assert.isNotNull(cloningStrategy);
30  		wrappedObject = objectToWrap;
31  		cloner = cloningStrategy;
32  	}
33  
34  	/**
35  	 * Indicates whether or not <code>null</code> is what is wrapped.
36  	 * 
37  	 * @return true if the wrapped "object" is actually <code>null</code>
38  	 */
39  	public boolean isNull() {
40  		return wrappedObject == null;
41  	}
42  
43  	/**
44  	 * Provides direct access to the wrapped object.
45  	 * 
46  	 * @return the actual wrapped object without any wrapping
47  	 */
48  	public Object getUnwrappedSelf() {
49  		return wrappedObject;
50  	}
51  
52  	/**
53  	 * Indicates whether or not the wrapped object is an instance of the type with the given name in the wrapped
54  	 * object's <code>ClassLoader</code>(s). It takes a parameter of type <code>String</code> instead of
55  	 * <code>Class</code> so that the test can be performed for <code>Class</code>es that do not have an equivalent
56  	 * in the caller's <code>ClassLoader</code>.
57  	 * 
58  	 * @param typeName the name of the type against which the wrapped object will be checked
59  	 * @return true if the wrapped object is an instance of the type with the given name in the wrapped object's
60  	 *         <code>ClassLoader</code>(s)
61  	 */
62  	public boolean isInstanceOf(String typeName) {
63  		Assert.isNotNull(typeName);
64  		return Transloader.DEFAULT.wrap(wrappedObject.getClass()).isAssignableTo(typeName);
65  	}
66  
67  	/**
68  	 * Gets an equivalent of the wrapped object with all <code>Class</code>es referenced being loaded from the given
69  	 * <code>ClassLoader</code>. Every object referenced in the object graph starting with the object returned will
70  	 * be able to be cast to its respective types in the given <code>ClassLoader</code>.
71  	 * <p>
72  	 * This implementation employs the <code>CloningStrategy</code> configured at construction. Note that using
73  	 * {@link CloningStrategy#MINIMAL} (which is not the default strategy in {@link Transloader#DEFAULT}) will often
74  	 * effect some changes within the object graph that starts with the wrapped object itself, as opposed to producing a
75  	 * completely new, seperate graph. Using {@link CloningStrategy#MAXIMAL} instead prevents this, producing a purely
76  	 * seperate clone without any changes within the wrapped object graph, at the cost of potentially far greater
77  	 * cloning effort. An object graph altered by cloning with {@link CloningStrategy#MINIMAL} can of course be restored
78  	 * entirely for use with other objects of <code>Class</code>es from its original <code>ClassLoader</code>(s)
79  	 * by cloning it back with those original <code>ClassLoader</code>(s), but this is an extra coding step and
80  	 * somewhat reduces the effort saved by not using {@link CloningStrategy#MAXIMAL} in the first place.
81  	 * </p>
82  	 * 
83  	 * @param classLoader the <code>ClassLoader</code> to use in creating an equivalent of the wrapped object
84  	 * @return an equivalent of the wrapped object with all <code>Class</code>es referenced being loaded from the
85  	 *         given <code>ClassLoader</code>
86  	 */
87  	public Object cloneWith(ClassLoader classLoader) {
88  		Assert.isNotNull(classLoader);
89  		if (isNull()) return null;
90  		try {
91  			return cloner.cloneObjectUsingClassLoader(getUnwrappedSelf(), classLoader);
92  		} catch (Exception e) {
93  			throw new TransloaderException("Unable to clone '" + getUnwrappedSelf() + "'.", e);
94  		}
95  	}

97  	/**
98  	 * Invokes on the wrapped object the method described by the given invocation description, with the parameters given
99  	 * by the same. Finds the method reflectively using parameter types loaded from the wrapped object's
100 	 * <code>ClassLoader</code>(s). Any parameters which refer to <code>Class</code>es that are foreign to the
101 	 * wrapped object's <code>ClassLoader</code>(s) are cloned using the <code>CloningStrategy</code> injected at
102 	 * construction.
103 	 * 
104 	 * @param description the description of the invocation to be performed
105 	 * @return the result of performing the invocation described by <code>description</code>
106 	 */
107 	public Object invoke(InvocationDescription description) {
108 		Assert.isNotNull(description);
109 		try {
110 			Class wrappedClass = getUnwrappedSelf().getClass();
111 			// TODO collect all ClassLoaders from the object graph into an abstraction named CollectedClassLoader
112 			ClassLoader wrappedClassLoader = wrappedClass.getClassLoader();
113 			Class[] parameterTypes = ClassWrapper.getClasses(description.getParameterTypeNames(), wrappedClassLoader);
114 			Object[] clonedParameters =
115 					(Object[]) cloner.cloneObjectUsingClassLoader(description.getParameters(), wrappedClassLoader);
116 			Method method = wrappedClass.getMethod(description.getMethodName(), parameterTypes);
117 			return method.invoke(getUnwrappedSelf(), clonedParameters);
118 		} catch (Exception e) {
119 			// TODO test Exception from invoke
120 			throw new TransloaderException(
121 					"Unable to invoke '" + description.getMethodName() + Arrays.asList(description.getParameterTypeNames()) + "' on '" + getUnwrappedSelf() + "'.",
122 					e);
123 		}
124 	}

126 	/**
127 	 * Makes an implementation of the given <code>interface</code> that calls through to the wrapped object. This is
128 	 * particularly useful if you have access in the current <code>ClassLoader</code> to an <code>interface</code>
129 	 * that the wrapped object is expected to implement, except that it actually implements the equivalent from a
130 	 * different <code>ClassLoader</code>. It is therefore usefully employed in conjunction with
131 	 * {@link #isInstanceOf(String)}.
132 	 * <p>
133 	 * This method will not fail fast if the wrapped object does not implement its own <code>ClassLoader</code>'s
134 	 * equivalent to the given <code>interface</code>, so can also be used for "duck"-typing, as a more syntactically
135 	 * elegant alternative to using {@link #invoke(InvocationDescription)}, if desired.
136 	 * </p>
137 	 * 
138 	 * @param targetInterface the <code>interface</code> that the returned object can be cast to
139 	 * @return a {@link Proxy} to the wrapped object that implements <code>desiredInterface</code>
140 	 */
141 	public Object makeCastableTo(Class targetInterface) {
142 		Assert.isNotNull(targetInterface);
143 		// TODO collect all ClassLoaders from the object graph into an abstraction named CollectedClassLoader
144 		return Proxy.newProxyInstance(getClass().getClassLoader(), new Class[] {targetInterface}, new Invoker());
145 	}
146 
147 	private class Invoker implements InvocationHandler {
148 		public Object invoke(Object proxy, Method method, Object[] parameters) throws Throwable {
149 			return ObjectWrapper.this.invoke(new InvocationDescription(method, parameters));
150 		}
151 	}
