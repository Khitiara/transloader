9   /**
10   * A utility for traversing through entire object graphs which may contain cyclic references, without thereby entering
11   * an endless loop. The implementation is thread-safe.
12   * 
13   * @author Jeremy Wales
14   */
15  public final class CyclicReferenceSafeTraverser {
16  	private final ThreadLocal referenceHistoryForThread = new ThreadLocal();
17  
18  	/**
19  	 * Executes the given the traversal over the current location in the object graph if it has not already been
20  	 * traversed in the current journey through the graph.
21  	 * 
22  	 * @param traversal the traversal to execute
23  	 * @param currentObjectInGraph the location in the object graph over which to perform the traversal
24  	 * @return the result of performing the traversal
25  	 * @throws Exception can throw any <code>Exception</code> from the traversal itself
26  	 */
27  	public Object performWithoutFollowingCircles(Traversal traversal, Object currentObjectInGraph) throws Exception {
28  		Assert.areNotNull(traversal, currentObjectInGraph);
29  		Map referenceHistory = getReferenceHistory();
30  		if (referenceHistory.containsKey(currentObjectInGraph)) return referenceHistory.get(currentObjectInGraph);
31  		referenceHistory.put(currentObjectInGraph, null);
32  		Object result = traversal.traverse(currentObjectInGraph, referenceHistory);
33  		// TODO make it so removals can only happen here
34  		// TODO test drive putting the removal in a finally block to prevent memory leaks following Exceptions
35  		referenceHistory.remove(currentObjectInGraph);
36  		return result;
37  	}
38  
39  	private Map getReferenceHistory() {
40  		Map referenceHistory = (Map) referenceHistoryForThread.get();
41  		if (referenceHistory == null) {
42  			referenceHistoryForThread.set(referenceHistory = new IdentityMap());
43  		}
44  		return referenceHistory;
45  	}
46  
47  	/**
48  	 * The callback interface for implementing a traversal over an object graph.
49  	 * 
50  	 * @author Jeremy Wales
51  	 */
52  	public static interface Traversal {
53  		/**
54  		 * Performs some logic on the given current location in the object graph. May update the given reference history
55  		 * to associate the given current object with the result of traversing it so that the result can be reused next
56  		 * time the same object is encountered in the same journey over the graph.
57  		 * 
58  		 * @param currentObjectInGraph the location in the object graph over which to perform the traversal
59  		 * @param referenceHistory the history of objects already traversed and the results of traversing them
60  		 * @return the result of traversing <code>currentObjectInGraph</code>
61  		 * @throws Exception can throw any <code>Exception</code> depending on the implementation
62  		 */
63  		Object traverse(Object currentObjectInGraph, Map referenceHistory) throws Exception;
64  	}
