83  	public static Object anyInstanceOf(Class type) {
84  		try {
85  			if (type == null || type == void.class) return null;
86  			if (type.isArray()) return anyArrayOf(type.getComponentType());
87  			Object triangulatedInstance = tryToTriangulateFromThisClass(type);
88  			if (triangulatedInstance != null) return triangulatedInstance;
89  			if (type.isInterface())
90  				return Proxy.newProxyInstance(Thread.currentThread().getContextClassLoader(), new Class[] {type},
91  						new TriangulatingInvocationHandler());
92  			if (Modifier.isAbstract(type.getModifiers()))
93  				return Enhancer.create(type, new TriangulatingInvocationHandler());
94  			return ObjenesisHelper.newInstance(type);
95  		} catch (Exception e) {
96  			throw new NestableRuntimeException(e);

109 	private static Object tryToTriangulateFromThisClass(Class type) throws Exception {
110 		for (int i = 0; i < MY_METHODS.length; i++) {
111 			Method method = MY_METHODS[i];
112 			Class returnType = method.getReturnType();
113 			boolean hasNoParameters = method.getParameterTypes() == null || method.getParameterTypes().length == 0;
114 			if (returnType == type && hasNoParameters) {
115 				return method.invoke(null, new Object[0]);
116 			}
117 		}
118 		return null;
119 	}

123 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
124 			if (method.getReturnType() == Void.class) return null;
125 			if (method.getName().equals("equals") && method.getParameterTypes().length == 1 && method.getParameterTypes()[0] == Object.class)
126 				return new Boolean(proxy == args[0]);
127 			if (method.getName().equals("hashCode") && method.getParameterTypes().length == 0)
128 				return new Integer(System.identityHashCode(proxy));
129 			if (method.getName().equals("toString") && method.getParameterTypes().length == 0)
130 				return TriangulatingInvocationHandler.class.getName() + '#' + System.identityHashCode(proxy);
131 			return anyInstanceOf(method.getReturnType());
132 		}
