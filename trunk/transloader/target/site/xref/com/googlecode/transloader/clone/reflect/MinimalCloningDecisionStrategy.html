11  /**
12   * When injected into a {@link ReflectionCloningStrategy}, decides that only those objects whose <code>Class</code>es
13   * would be different if loaded through the other <code>ClassLoader</code> should be cloned.
14   * 
15   * @author Jeremy Wales
16   */
17  public final class MinimalCloningDecisionStrategy implements CloningDecisionStrategy {
18  	// TODO test BigInteger and BigDecimal
19  	private static final List KNOWN_SHARED_IMMUTABLES =
20  			Arrays.asList(new Class[] {String.class, BigInteger.class, BigDecimal.class});
21  
22  	/**
23  	 * Decides that the given object should be shallow copied if its <code>Class</code> would be different when loaded
24  	 * through the given <code>ClassLoader</code>.
25  	 * 
26  	 * @param original the candidate for cloning
27  	 * @param targetClassLoader the <code>ClassLoader</code> it may be cloned with
28  	 * @return <code>true</code> if <code>original</code>'s <code>Class</code> would be different when loaded
29  	 *         through <code>targetClassLoader</code>
30  	 */
31  	public boolean shouldCloneObjectItself(Object original, ClassLoader targetClassLoader) {
32  		Assert.areNotNull(original, targetClassLoader);
33  		return !isSameInClassLoader(original.getClass(), targetClassLoader);
34  	}
35  
36  	private boolean isSameInClassLoader(Class originalClass, ClassLoader targetClassLoader) {
37  		return originalClass.equals(ClassWrapper.getClass(originalClass.getName(), targetClassLoader));
38  	}
39  
40  	/**
41  	 * Decides to not even consider cloning the objects referenced by the given object if it is known to an immutable
42  	 * object shared of a type shared among all <code>ClassLoader</code>s e.g. primitive wrappers and
43  	 * <code>String</code>s.
44  	 * 
45  	 * @param original the candidate for cloning
46  	 * @param targetClassLoader the <code>ClassLoader</code> it may be cloned with; ignored in this implementation
47  	 * @return <code>true</code> unless <code>original</code>'s <code>Class</code> makes it a known immutable of
48  	 *         type shared among all <code>ClassLoader</code>s
49  	 */
50  	public boolean shouldCloneObjectContent(Object original, ClassLoader targetClassLoader) {
51  		Assert.areNotNull(original, targetClassLoader);
52  		return !isEffectivelyPrimitive(original.getClass());
53  	}
54  
55  	private boolean isEffectivelyPrimitive(Class originalClass) {
56  		return FieldReflector.PRIMITIVE_WRAPPERS.contains(originalClass) || KNOWN_SHARED_IMMUTABLES.contains(originalClass);
57  	}
58  }
