8   /**
9    * The API by which an object can be cloned using a different <code>ClassLoader</code> than that/those which loaded
10   * the <code>Class</code>es it references. It can be used directly or serves as the strategy interface for
11   * customising the behaviour of {@link com.googlecode.transloader.ObjectWrapper}s.
12   * 
13   * @author Jeremy Wales
14   */
15  public interface CloningStrategy {
16  	/**
17  	 * The implementation which clones as little as possible to make the given object graph use <code>Class</code>es
18  	 * that are the same as those that would be loaded through the given <code>ClassLoader</code>. This is the most
19  	 * efficient implementation to use simply for the purpose of making sure the returned object graph is perfectly
20  	 * compatible with all other objects referencing <code>Class</code>es loaded through the given
21  	 * <code>ClassLoader</code>. It is also the implementation that is most likely to succeed in cloning any given
22  	 * object graph because by attempting to clone as few objects as possible it is less likely to attempt to clone an
23  	 * object that cannot be cloned (e.g. a non-{@link java.io.Serializable} with <code>final</code> fields in a
24  	 * pre-Java-5 JVM).
25  	 * <p>
26  	 * However, the fact that it only clones what is necessary to make the object graph compatible with the given
27  	 * <code>ClassLoader</code> means that usually <b>not all</b> of the objects in the graph will be cloned. This
28  	 * means that, depending on the <code>Class</code>es in the object graph and which of these is the same if loaded
29  	 * through the given <code>ClassLoader</code>, it is possible that a top level object is not cloned but objects
30  	 * it references through fields are cloned. This effectively means that the existing object graph can be altered
31  	 * rather than a new, purely seperate object graph being created. This <b>may not be what you want</b> if you want
32  	 * to continue using the given object in its original context. In which case, use {@link CloningStrategy#MAXIMAL}.
33  	 * </p>
34  	 * 
35  	 * @see ReflectionCloningStrategy
36  	 * @see MinimalCloningDecisionStrategy
37  	 */
38  	CloningStrategy MINIMAL =
39  			new ReflectionCloningStrategy(new MinimalCloningDecisionStrategy(), new ObjenesisInstantiationStrategy(),
40  					new SerializationCloningStrategy());
41  
42  	/**
43  	 * The implementation which clones every <code>Object</code> in the given object graph. The given object graph is
44  	 * not altered in any way and a completely new copy of the object graph referencing only <code>Class</code>es
45  	 * loaded through the given <code>ClassLoader</code> is returned. Only primitives, which are not
46  	 * <code>Object</code>s, are not cloned because they cannot be, as there is no concept of different references to
47  	 * the same primitive value in Java. Similiar in behaviour to {@link SerializationCloningStrategy} except that it
48  	 * can clone more object graphs because it does not rely on all referenced objects being
49  	 * {@link java.io.Serializable} and also performs much faster than serialization.
50  	 * 
51  	 * @see ReflectionCloningStrategy
52  	 * @see MaximalCloningDecisionStrategy
53  	 */
54  	CloningStrategy MAXIMAL =
55  			new ReflectionCloningStrategy(new MaximalCloningDecisionStrategy(), new ObjenesisInstantiationStrategy(),
56  					new SerializationCloningStrategy());
57  
58  	/**
59  	 * Clones the given object using the given <code>ClassLoader</code>.
60  	 * 
61  	 * @param original the original object to be cloned
62  	 * @param targetClassLoader the <code>ClassLoader</code> by which to load <code>Class</code>es for clones
63  	 * @return the result of cloning the object graph
64  	 * @throws Exception can throw any <code>Exception</code> depending on the implementation
65  	 */
66  	Object cloneObjectUsingClassLoader(Object original, ClassLoader targetClassLoader) throws Exception;
67  }
