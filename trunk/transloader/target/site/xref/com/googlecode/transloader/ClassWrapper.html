60  	/**
61  	 * Indicates whether or not the wrapped <code>Class</code> is assignable to a <code>Class</code> with the given
62  	 * name. It takes a parameter of type <code>String</code> instead of <code>Class</code> so that the test can be
63  	 * performed for <code>Class</code>es that do not have an equivalent in the caller's <code>ClassLoader</code>.
64  	 *

67  	 */
68  	public boolean isAssignableTo(String typeName) {
69  		Assert.isNotNull(typeName);
70  		return classIsAssignableToType(wrappedClass, typeName);
71  	}

73  	/**
74  	 * Loads the <code>Class</code> with the given name from the given <code>ClassLoader</code>.
75  	 * 
76  	 * @param className the name of the <code>Class</code>
77  	 * @param classLoader the <code>ClassLoader</code> with which to load it
78  	 * @return the <code>Class</code> with the given name loaded from the given <code>ClassLoader</code>
79  	 * @throws TransloaderException if the <code>Class</code> cannot be found in the given <code>ClassLoader</code>
80  	 */
81  	public static Class getClass(String className, ClassLoader classLoader) {
82  		Assert.areNotNull(className, classLoader);
83  		try {
84  			return ClassUtils.getClass(classLoader, className, false);
85  		} catch (ClassNotFoundException e) {
86  			// TODO test ClassNotFoundException
87  			throw new TransloaderException(
88  					"Unable to load Class '" + className + "' from ClassLoader '" + classLoader + "'.", e);
89  		}
90  	}

92  	/**
93  	 * Loads the <code>Class</code>es with the given names from the given <code>ClassLoader</code>.
94  	 * 
95  	 * @param classNames the names of the <code>Class</code>es
96  	 * @param classLoader the <code>ClassLoader</code> with which to load them
97  	 * @return the <code>Class</code>es with the given names loaded from the given <code>ClassLoader</code>
98  	 * @throws TransloaderException if even one of the <code>Class</code>es cannot be found in the given
99  	 *             <code>ClassLoader</code>
100 	 */
101 	public static Class[] getClasses(String[] classNames, ClassLoader classLoader) {
102 		Assert.areNotNull(classNames, classLoader);
103 		Class[] classes = new Class[classNames.length];
104 		for (int i = 0; i < classes.length; i++) {
105 			classes[i] = getClass(classNames[i], classLoader);
106 		}
107 		return classes;
108 	}

110 	private static boolean classIsAssignableToType(Class rootClass, String typeName) {
111 		List allClasses = new ArrayList();
112 		allClasses.add(rootClass);
113 		allClasses.addAll(ClassUtils.getAllSuperclasses(rootClass));
114 		allClasses.addAll(ClassUtils.getAllInterfaces(rootClass));
115 		return ClassUtils.convertClassesToClassNames(allClasses).contains(typeName);
116 	}
